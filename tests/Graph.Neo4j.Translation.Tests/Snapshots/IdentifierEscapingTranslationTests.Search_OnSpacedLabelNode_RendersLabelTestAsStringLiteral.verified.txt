== Cypher ==
CALL db.index.fulltext.queryNodes('node_fulltext_index', $p0) YIELD node AS n
WHERE NOT (EXISTS { MATCH ()-[__complexRootOwner]->(n) WHERE __complexRootOwner.__graphModelComplexProperty = true }) AND 'Label With Space' IN labels(n)
RETURN {
                Node: n,
                ComplexProperties: []
            } AS Node

== Parameters ==
p0 = "alice"
