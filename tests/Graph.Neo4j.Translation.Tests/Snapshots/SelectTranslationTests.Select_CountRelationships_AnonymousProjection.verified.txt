== Cypher ==
MATCH (src:Person|Manager)
WHERE NOT (EXISTS { MATCH ()-[__complexRootOwner]->(src) WHERE __complexRootOwner.__graphModelComplexProperty = true })
RETURN src.FirstName AS FirstName, COUNT { MATCH (src)-[:KNOWS]->() } AS Outgoing, COUNT { MATCH (src)<-[:KNOWS]-() } AS Incoming, COUNT { MATCH (src)-[:KNOWS]-() } AS Total
