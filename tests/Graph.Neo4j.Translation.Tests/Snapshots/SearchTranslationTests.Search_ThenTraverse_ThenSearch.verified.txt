== Cypher ==
CALL db.index.fulltext.queryNodes('node_fulltext_index', $p0) YIELD node AS n
MATCH (n)-[r:KNOWS]->(tgt:Person|Manager)
CALL db.index.fulltext.queryNodes('node_fulltext_index', $p1) YIELD node AS searchedNode
WHERE NOT (EXISTS { MATCH ()-[__complexRootOwner]->(n) WHERE __complexRootOwner.__graphModelComplexProperty = true }) AND ('Person' IN labels(n) OR 'Manager' IN labels(n)) AND tgt = searchedNode AND ('Person' IN labels(searchedNode) OR 'Manager' IN labels(searchedNode)) AND r.__graphModelComplexProperty IS NULL
OPTIONAL MATCH src_path = (tgt)-[rels*1..5]->(prop)
WHERE ALL(rel IN rels WHERE rel.__graphModelComplexProperty = true)
WITH tgt, n, r,
    CASE WHEN src_path IS NULL THEN [] ELSE [i IN range(0, size(rels) - 1) | {
        ParentNode: CASE WHEN i = 0 THEN tgt ELSE nodes(src_path)[i] END,
        Relationship: rels[i],
        SequenceNumber: rels[i].SequenceNumber,
        Property: nodes(src_path)[i + 1]
    }] END AS src_property_path
WITH tgt, n, r, reduce(flat = [], path IN collect(src_property_path) | flat + path) AS src_properties
RETURN {
    Node: tgt,
    ComplexProperties: src_properties
} AS Node

== Parameters ==
p0 = "alice"
p1 = "bob"
