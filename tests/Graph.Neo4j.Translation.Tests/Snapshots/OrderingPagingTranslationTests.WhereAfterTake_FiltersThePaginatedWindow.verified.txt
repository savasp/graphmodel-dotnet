== Cypher ==
MATCH (src:Person|Manager)
WITH *
ORDER BY src.Age
LIMIT 5
WITH *
WHERE src.Age >= $p0
WITH *,
    src.Age AS __projectionOrder0
OPTIONAL MATCH src_path = (src)-[rels*1..5]->(prop)
WHERE ALL(rel IN rels WHERE rel.__graphModelComplexProperty = true)
WITH src, __projectionOrder0,
    CASE WHEN src_path IS NULL THEN [] ELSE [i IN range(0, size(rels) - 1) | {
        ParentNode: CASE WHEN i = 0 THEN src ELSE nodes(src_path)[i] END,
        Relationship: rels[i],
        SequenceNumber: rels[i].SequenceNumber,
        Property: nodes(src_path)[i + 1]
    }] END AS src_property_path
WITH src, __projectionOrder0, reduce(flat = [], path IN collect(src_property_path) | flat + path) AS src_properties
WITH {
    Node: src,
    ComplexProperties: src_properties
} AS Node, __projectionOrder0
ORDER BY __projectionOrder0
RETURN Node

== Parameters ==
p0 = 21
