== Cypher ==
MATCH (src:Person|Manager)
CALL db.index.fulltext.queryNodes('node_fulltext_index', $p0) YIELD node AS n
WHERE ('Person' IN labels(n) OR 'Manager' IN labels(n)) AND n.Age > $p1

            // All complex property paths from src
            OPTIONAL MATCH src_path = (n)-[rels*1..]->(prop)
            WHERE ALL(rel in rels WHERE type(rel) STARTS WITH '__PROPERTY__')
            WITH n,
                CASE 
                    WHEN src_path IS NULL THEN []
                    ELSE [i IN range(0, size(rels)-1) | {
                        ParentNode: 
                            CASE 
                                WHEN i = 0 THEN n
                                ELSE nodes(src_path)[i]
                            END,
                        Relationship: rels[i],
                        SequenceNumber: rels[i].SequenceNumber,
                        Property: nodes(src_path)[i+1]
                    }]
                END AS src_flat_property

            WITH n,
                reduce(flat = [], l IN collect(src_flat_property) | flat + l) AS src_flat_properties
            WITH n, apoc.coll.toSet(src_flat_properties) AS src_flat_properties


            RETURN {
                Node: n,
                ComplexProperties: src_flat_properties
            } AS Node

== Parameters ==
p0 = "Alice"
p1 = 21
